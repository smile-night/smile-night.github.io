1.冒泡排序
```
    const maopao = (arr) => {
      let temp = 0;
      let i = 0;
      while(i < arr.length - 1) {
        for (let j = 0; j + 1 <= arr.length - 1 - l && i < arr.length - 1; j++ ) {
          if (arr[j] > arr[j + 1]) {
            temp = arr[j];
            arr[j] = arr[j + 1];
            arr[j + 1] = temp;
          }
        }
        ++i;
      }
      console.log('冒泡算法', arr);
      return arr;
    }
```

2.直接插入排序
```
	const insertSort = (arr) => {
      if (arr.length < 2) {
        return arr;
      }
      let temp = 0;
      for(let i = 1; i < arr.length; i++) {
        temp = arr[i];
        for(var j = i - 1 ; arr[j] > temp && j >= 0; j--) {
          arr[j + 1] = arr[j];
          if (j == 0 ||  arr[j] >= temp) {
            arr[j] = temp;
          }
        }
      }
      return arr;
  }
```
